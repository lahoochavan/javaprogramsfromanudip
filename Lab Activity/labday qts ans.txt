1. Write a Java Program to check given number is even or odd.


public class ifstatement {

	public static void main(String[] args) {
		int n = 24;
		if (n % 2 == 0) {
			System.out.println("The number is even : "+ n);
		 } else {
			System.out.println("The number is odd : "+ n);
		 }
	}

}
